void show_effect() {

    }
    else if (input == '=' || input == '+' || input == '-' || input == 8 || input == 'a' || input == 'A' || input == 'C' || input == 'c' || input == '*' || input == '/') {

void hapche() {

    int su = 0;

    //합체
    for (l = 0; l < i; l++) {

        su = su + (arr[l] - 48) * pow(10, (i - l - 1));

    }

    i = 0;//버퍼비우기

    //brr에 삽입.
    if (brr[0]) {//값이 이미 존재할경우
        if (flag == 1) {//새출발:
            brr[0] = su;
            orr[0] = 0;//next op 들어왔을때 orr[0]의= 연산 수행되서 출력되는것 방지. next op가 orr[0] 에 들어갈 수 있도록. 
            flag = 0;
        }
        else {
            brr[1] = su;
        }

    }
    else {

        brr[0] = su;
    }


}

void operation() {

    int temp;
    int count = 0;

    hapche();

    if (orr[0]) {
        if (orr[0] == '=') {
            orr[0] = input;//=으로 결과 출력 후 이어서 연산: next op 들어왔을때 orr[0]의= 연산 수행되서 출력되는것 방지. next op가 orr[0] 에 들어갈 수 있도록. 
        }

        orr[1] = input;
    }
    else {
        orr[0] = input;//최초실행.

    }


    if (orr[1]) {//+_*/= 에서 두번쨰 연산자가 들어오면 첫번째 연산자 실행.

        if (orr[0] == '+') {
            brr[0] = brr[0] + brr[1];
            brr[1] = 0;

        }
        else if (orr[0] == '-') {
            brr[0] = brr[0] - brr[1];
            brr[1] = 0;
        }

        else if (orr[0] == '/') {
            if (brr[0] == 0) {
                system("cls");
                set_ready();

                gotoxy(2, 2); printf("-------ERROR!-------");
                Sleep(2500);
                reset();

                system("cls");
                set_ready();
            }
            else {
                brr[0] = brr[0] / brr[1];
                brr[1] = 0;
            }
        }
        else if (orr[0] == '*') {
            brr[0] = brr[0] * brr[1];
            brr[1] = 0;

        }
        else if (orr[0] == '=') {
            temp = brr[0];

            while (temp != 0) {
                temp = temp / 10;
                count++;
            }
            if (brr[0] == 0) {//0이면 1자리수이지만 while문을 실행하지 않으므로
                set_start_point();
            }
            else {
                gotoxy(19 - count + 1, 2);
            }
            printf("%d", brr[0]);
            gotoxy(19, 3); printf(" ");


        }

        orr[0] = orr[1];
        orr[1] = '0';
    }
}

void startCalc() {


    while (1) {

        set_start_point();

        input = _getche();

        show_effect();


        //숫자
        if ('0' <= input && input <= '9') {

            arr[i] = input;//입력값 저장

            if (orr[0] == '=') {//=으로 연산 결과를 내고 그다음 바로 숫자를 입력한 경우. 새출발      
                flag = 1;
            }


            if (i != 0) {//기존값이 있다면 출력. (입력받은거 제외하고)

                gotoxy(19 - i, 2);
                for (k = 0; k < i; k++) {
                    printf("%c", arr[k]);
                }
            }

            i++;//arr index 증가. max 9

            if (i >= 10) {// 안전장치.

                system("cls");
                set_ready();

                gotoxy(2, 2); printf("ERROR! PRESS NUMBER");
                reset();

                continue;

            }

            continue;
            disp_proc();
        }
        //연산자
        else if (input == '=' || input == '+' || input == '-' || input == 8 || input == 'a' || input == 'A' || input == 'C' || input == 'c' || input == '*' || input == '/') {

            switch (input) {

            case '+':
            case '-':
            case '*':
            case '/':
            case '=':
                show_effect();
                operation();
 
                continue;

            case 'A': //C. all clear
            case 'a':
                show_effect();
                reset();
                continue;

            case 'C'://CE. orr에 안들어감
            case 'c':
                show_effect();
                if (orr[0]) {
                    i = 0;
                }
                else {
                    brr[0] = 0;
                    continue;

            case 8:
                show_effect();


                continue;

            default:
                continue;

                }

            }
        }
        else if (input == 'x') {//프로그램 종료 x


            gotoxy(17, 2); printf("OFF");
            gotoxy(0, 20);
            exit(0);

        }
        //기타 문자
        else {
            system("cls");
            set_ready();

            gotoxy(2, 2); printf("-------ERROR!-------");
            Sleep(2500);
            reset();

            system("cls");
            set_ready();
            continue;

        }


    }




}
